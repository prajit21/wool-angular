Customizer configuration

To replace our design with your desired theme. Please do configuration as mention

Path : src > app > shared > services > layout.service.ts 

          
public config = {
  settings: {
      layout_type: '{{layout.config.settings.layout_type}}',
      layout_version: '{{layout.config.settings.layout_version}}',
      sidebar_type: '{{layout.config.settings.sidebar_type}}',
  },
  color: {
      primary_color: '{{layout.config.color.primary}}',
      secondary_color: '{{layout.config.color.secondary}}'
  }
}
